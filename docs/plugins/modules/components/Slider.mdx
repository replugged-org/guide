import CSSProperties from "@site/src/components/CSSProperties";

### Example

```tsx
import { components } from "replugged";
const { SliderItem } = components;

export function Settings(): React.ReactElement {
  return (
    <SliderItem
      {...util.useSetting(cfg, "foo", 0)}
      minValue={0}
      maxValue={100}
      markers={[0, 25, 50, 75, 100]}>
      Slider title
    </SliderItem>
  );
}
```

### Enums

| Enum                     | Description                                      |
| ------------------------ | ------------------------------------------------ |
| `Slider.MarkerPositions` | Slider marker positions <br />`ABOVE` \| `BELOW` |
